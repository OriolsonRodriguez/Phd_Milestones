  \caption{Notation used throughout the document}
  \label{tab:notation}

    \toprule
    \textbf{Symbol} & \textbf{Meaning} \\
    \midrule
    $N$             & Set of treaty parties (agents), $|N| = n$ \\
    $i, j \in N$    & Individual state actors \\
    $s_i \in S$     & Strategy of party $i$ \\
    $u_i$           & Utility function of party $i$ \\
    $\alpha_{ij}$   & Affinity score between parties $i$ and $j$ \\
    $\beta_{ij}$    & Aversion score between parties $i$ and $j$ \\
